# Triggers when the game starts
on_game_start = {
	events = {
		ps.1 # Spawn Planet-States
	}
}

# A fleet has entered a system
# Scope = Fleet
# From = System
on_entering_system_fleet = {
	events = {
		ps.12 # Entering Planet-State system for the first time
	}
}

# No scope, like on_game_start
on_yearly_pulse = {
	events = {
		ps.10 # Planet-State Yearly Border Opinion
	}
}

# Triggers when trying to open the diplomacy view for countries with custom diplomacy.
# Country scope
# This = target country (player)
# From = source country
on_custom_diplomacy = {
	events = {
		ps.100 # The Egalitarian
		#ps.200 # The Authoritarian
		#ps.300 # The Militarist
		#ps.400 # The Pacifist
	}
}

#This = destroyed country
#From = optional, destroyer (country)
on_country_destroyed = {
	events = {
		ps.1000 # Egalitarian PS destroyed (HIDDEN)
		#ps.2000
		#ps.3000
		#ps.4000
	}
}
